TreePanel

Example of a basic TreePanel:

	var tree = new Ext.tree.TreePanel({
		title: 'Categories',
		renderTo: document.body,
		loader: new Ext.tree.TreeLoader({
			url: 'sample.json'
		}),
		root: new Ext.tree.AsyncTreeNode({
		    text: 'Categories',
			id: 'categories',
			expanded: true
		})
	});
